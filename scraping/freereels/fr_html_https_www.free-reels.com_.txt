<script>
    function alertMsg(msg){
        $("#submitErr").text(msg);
        $("#submitErr").show();
    }
    function checkMailFormat(email){
        var emailRegExp = /^\w(\.|\w)*@(\.|\w|\-)+\.(\.|\w)*\w$/;
        return emailRegExp.test(email);
    }
    function checkLogin(email, password){
        if(email == ''){
            alertMsg("Email is required");
            return false;
        }
        if(password == ''){
            alertMsg("Password is required");
            return false;
        }
        if(!checkMailFormat(email)){
            alertMsg("Email doesn't match the format");
            return false;
        }
        return true;
    }
    function checkRegister(email, name, password, password_again){
        if(email == ''){
            alertMsg("Email is required");
            return false;
        }
        if(name == ''){
            alertMsg("Name is required");
            return false;
        }
        if(password == '' || password_again == ''){
            alertMsg("Password is required");
            return false;
        }
        if(password != password_again) {
            alertMsg("The two passwords aren't the same");
            return false;
        }

        if(!checkMailFormat(email)){
            alertMsg("Email doesn't match the format");
            return false;
        }
        return true;
    }
    $(function(){
        var matches = window.location.href.match(/\?from=(.*)&?/i)
        if(matches != null && matches[1]=="register"){
            $("#signupSuc").show();
        }
        $("#login_submit").click(function () {
            $("#signupSuc").hide();
            $("#submitErr").hide();
            var email = $("#login [name='email']").val();
            var password = $("#login [name='password']").val();
            if(checkLogin(email, password)) {
                $.ajax({
                    url: "/login",
                    type: "POST",
                    data: {email: email, password: password},
                    success: function (data) {
                        window.location.pathname = "/";
                    },
                    error: function (xhr, status, error) {
                        alertMsg(xhr.responseText);
                    },
                });
            }
        });
        $("#register_submit").click(function () {
            $("#signupSuc").hide();
            $("#submitErr").hide();
            var email = $("#register [name='email']").val();
            var name = $("#register [name='name']").val();
            var password = $("#register [name='password']").val();
            var password_again = $("#register [name='password_again']").val();
            if(checkRegister(email, name, password, password_again)) {
                $.ajax({
                    url: "/register",
                    type: "POST",
                    data: {email: email, name: name, password:password},
                    success: function(data){
                        window.location.href="http://www.free-reels.com/login?from=register";
                    },
                    error: function(xhr, status, error) {
                        alertMsg(xhr.responseText);
                    },
                });
            }
        });
    });
</script>
